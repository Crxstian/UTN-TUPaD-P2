    public class TP2 {
        

    
    public static void main(String[] args) {
    
    Scanner scanner = new Scanner(System.in);
    double precioProducto,pesoKg;
    String zonaEnvio;
    System.out.print("Ingrese el precio del producto: ");
    precioProducto = scanner.nextDouble();
    System.out.print("Ingrese el peso del paquete en kg: ");
    pesoKg = scanner.nextDouble();
    scanner.nextLine();
    System.out.println("Ingrese la zona de envio (Nacional/Internacional): ");
    zonaEnvio = scanner.nextLine().toUpperCase();
    
    System.out.println("El costo de envio es: "+calcularCostoEnvio(pesoKg, zonaEnvio));
    System.out.println("El total a pagar es: "+calcularTotalCompra(precioProducto, calcularCostoEnvio(pesoKg, zonaEnvio)));

}


    public static double calcularCostoEnvio(double peso,String zona){
        double costoEnvio =0;
        if (zona.equalsIgnoreCase("Nacional")) {
            costoEnvio = peso*5;
        }else if(zona.equalsIgnoreCase("Internacional")){
            costoEnvio =  peso*10;
        }
        return costoEnvio;
    }
    public static double calcularTotalCompra(double  precioProducto,double costoEnvio){
        return precioProducto + costoEnvio;
    }    
    }
