    Scanner scanner = new Scanner(System.in);
    double precio,descuento,precioFinal;
    String categoria;
    System.out.println("Ingrese el precio del producto: ");
    precio = scanner.nextInt();
    scanner.nextLine(); 
    
    System.out.println("Ingrese la categoria del producto (A, B o C): ");
    categoria = scanner.nextLine().toUpperCase();
    
        switch (categoria) {
            case "A":
                System.out.println("Descuento aplicado: 10%");
                descuento = 0.10;
                break;
            case "B":
                System.out.println("Descuento aplicado: 15%");
                descuento = 0.15;
                break;
            case "C":
                System.out.println("Descuento aplicado: 20%");
                descuento = 0.20;
            default:
                System.out.println("Categoría inválida., Debe ingresar una de las opciones indicadas");
                return;
        }
    precioFinal = precio - (precio*descuento);
        System.out.println("Precio final: "+precioFinal);
